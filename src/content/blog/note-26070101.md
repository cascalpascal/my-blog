# Declaration & Initialization
변수를 담을 공간과 자료형 지정하는 것을 <code>선언</code>이라고 하고 그 공간에 값을 넣는 것을 <code>초기화</code>라고 한다.

### C
~~~ C
/* 선언 */
int x;
char c;
float f;

// 콤마 선언
int a, b;

// 포인터 선언
int *p;

// 구조체 선언
struct School *h;

/* 초기화 */
int x = 5;
char c = 'A';
float f = 3.16;

// 리셋
x = 10;

// 선언만 하고 초기화되지 않은 grabage value 출력
int num;
print("%d", num);

// 배열 초기화
int arrA[4] = {1, 3, 5, 7};
// 배열 크기 생략
int arrB[] = {2, 4, 6};
// arrC = {8, 9, 0, 0, 0}
int arrC[5] = {8, 9};
// arrD = {0, 0, 0}
int arrD[3] = {0}

// 구조체 초기화
struct Test {
    int score;
    int avg;
};
struct Test test1 = {90, 75};   // score = 90, avg = 75

// unsigned int: 음수 없이 0과 양수만 저장하는 정수 타입
struct Node {
    struct Node* next;
    unsigned int x;
};
// u는 unsigned 리터럴 접미사
struct Node n1 = {0, 5u};   // next = 0(NULL), x = 5
struct Node n2 = {0, 7u};   // next = 0(NULL), x = 7
~~~

### Java
~~~Java
/* 선언 */
int x;
char c;
double d;

// 콤마 선언
int a, b;

/* 초기화 */
int x = 5;
char c = 'A';
double = 3.16;

// 선언한 변수를 초기화하지 않으면 자동으로 기본값으로 설정(지역변수 제외)
int y;          // y = 0
long z;         // z = 0
double p;       // p = 0.0
float q;        // q = 0.0
boolean l;      // l = false
String name;    // name = null

class Test {
    // attribute(속성): 클래스 안, 메서드 밖
    int score;
    // parameter(매개변수): 메서드 소괄호 안
    void setScore(int x){
        // local variable(지역변수): 메서드 안
        int tmp = x;
        score = tmp;
    }
}

// 배열 초기화
int[] arrA = {1, 3, 5, 7, 9}
int[] arrB = new int[3];    // arrB = {0, 0, 0}
~~~

### Python
~~~Python
# Python에서는 타입을 지정하지 않아도 값에 따라 변수가 자동으로 생성됨

x = 5;          # int
y = "hello";    # str
z = 3.16;       # float

# 콤마 선언
a, b, c = 1, 2, 3;

# 배열 초기화
arrA = [1, 3, 5, 7, 9]
arrB = [0] * 3          // arrB = [0, 0, 0]
~~~
